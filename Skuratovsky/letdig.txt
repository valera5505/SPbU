#include <iostream>
using namespace std;

int main() {
	char ch;
	cin>>ch;
	if (isalpha(ch)) cout<<"ch is a letter\n"; else if (isdigit(ch)) cout<<"ch is a digit\n";
	try {
		if (!(isalpha(ch)) && !(isdigit (ch))) throw 0;}
		catch (int c)
		{if (!c) cout<<"ch is neither a digit nor a letter\n";
		}
		
	// your code goes here
	return 0;
}
