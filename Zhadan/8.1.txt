#include <iostream>
#include <fstream>
#include <string>


using namespace std;

void lol()
{

	string bufer;

	ifstream in("input.txt");
	ofstream out("output2.txt", ios_base::app);


	while (!in.eof())
	{
		out << endl;
		getline(in,bufer);
		out << bufer;
	}
	out << endl;
	in.close();
	out.close();

}

int main()
{
	string bufer1;
	ifstream in("output.txt");
	ofstream out("output2.txt");
	
	
	while (!in.eof())
	{
		getline(in, bufer1);
		if (bufer1.find("#include input") <= 255)
		{
			for (int i = 0; i < bufer1.find("#include input") + 14; i++) out << bufer1[i];
			out << endl;
			lol(); 
			out.close();
			out.open("output2.txt", ios_base::app);	
			bufer1.erase(0, bufer1.find("#include input") + 14);
			out << bufer1 << endl;

		}
		else out << bufer1 << endl;
	}
		
    


	in.close();
	out.close();
	system("pause");
	return 0;
}
